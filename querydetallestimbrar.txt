SELECT dn.total1,dn.total2,e.rfc,CONCAT(e.apaterno," ",e.amaterno," ",e.nombre) as nombre, dn.fechai,dn.fechaf,pn.fechapago,dn.sindicato,dn.puesto,e.fecha_ingreso,e.nss,e.clave,e.curp,e.jornada,dn.contrato,dn.producto,dn.id,dn.movimiento,p.descripcion,dn.unidad,dn.clavep,dn.banco,dn.cuenta_bancaria,dn.actividad,dn.proyecto,dn.partida,dn.clavepago,dn.clue,e.ingreso_acumulable,e.ingreso_no_acumulable, e.ultimo_sueldo from detalle_nomina dn, empleados e, puestos p,producto_nomina pn where dn.clave=e.clave and dn.rfc=e.rfc and dn.puesto=p.id_puestos AND pn.clave=dn.producto;
